set.seed(1903)
rows = 500
z1 = rnorm(rows)
z2 = rnorm(rows)
z3 = rnorm(rows)

y1 = 1 + z1
y2 = 5 + 2 * z1 + z2
df = data.frame(Y1 = y1, Y2 = y2)

– 3.a

df_1 = df
a = 2
b = 0
y2_rule = a * (y1 - 1) + b * (y2 - 5) + z3 < 0

y2_obs = y2[y2_rule]
y2_miss = y2[!y2_rule]

for (i in 1:rows) {
    if (y2_rule[i]) {
        df_1$Y2[i] = NA
    }
}

plot(density(y2_obs), col = "darkblue", xlab = "Y2", main = "MNAR")
lines(density(y2_miss), col = "red")
lines(density(df$Y2), col = "darkgreen")

    "red", "darkgreen"), lty = c(1, 1, 1), bty = "n")
